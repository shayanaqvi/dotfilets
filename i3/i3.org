* Modes
** Container layouts
Select from Tabbed, Stacked or Split layouts
#+begin_src
bindsym $mod+XF86LaunchA mode "$container-layout"; exec notify-send i3 "(t) tabbed\n(s) stacked\n(d) split"

mode "$container-layout" {
     bindsym t layout tabbed
     bindsym s layout stacking
     bindsym d layout toggle split
     bindsym Escape mode "default"; exec dunstctl close-all
     bindsym Return mode "default"; exec dunstctl close-all
     bindsym $mod+XF86LaunchA mode "default"; exec dunstctl close-all
}
#+end_src
** Resize
Resize windows in this mode
#+begin_src
bindsym $mod+r mode "resize"; exec dunstify i3 "Resize windows"
mode "resize" {
        bindsym h resize shrink width 2 px or 2 ppt
        bindsym j resize grow height 2 px or 2 ppt
        bindsym k resize shrink height 2 px or 2 ppt
        bindsym l resize grow width 2 px or 2 ppt

        bindsym Return mode "default"; exec dunstctl close-all
        bindsym Escape mode "default"; exec dunstctl close-all
        bindsym $mod+r mode "default"; exec dunstctl close-all
}
#+end_src
** Launch application
Launch an application by pressing a key
#+begin_src
bindsym $mod+XF86LaunchB mode "$launch-application"; exec dunstify i3 "Select an application to launch:\nMusikcube (m)\nRanger (e)\nNemo (E)\nEmacs (v)\nFirefox (f)"
mode "$launch-application" {
     bindsym m exec kitty musikcube; exec dunstctl close-all; mode "default"
     bindsym e exec kitty ranger; exec dunstctl close-all; mode "default"
     bindsym Shift+e exec nemo; exec dunstctl close-all; mode "default"
     bindsym v exec emacs; exec dunstctl close-all; mode "default"
     bindsym f exec firefox; exec dunstctl close-all; mode "default"
     bindsym Escape mode "default"; exec dunstctl close-all
     bindsym Return mode "default"; exec dunstctl close-all
     bindsym $mod+XF86LaunchB mode "default"; exec dunstctl close-all
}
#+end_src
** Exit session
Options for exiting the current session
#+begin_src
bindsym $mod+Control+q mode "$exit-session"; exec dunstify i3 "Logout (l)\nShutdown (s)\nReboot (r)"
mode "$exit-session" {
     bindsym l exec i3-msg exit
     bindsym s exec poweroff
     bindsym r exec reboot
     bindsym Escape mode "default"; exec dunstctl close-all
     bindsym Return mode "default"; exec dunstctl close-all
     bindsym $mod+Control+q mode "default"; exec dunstctl close-all
}
